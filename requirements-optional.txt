# Optional, heavier features — installed by setup.bat / setup.sh.
# Kept out of requirements.txt so a minimal build (no OCR, no translation)
# stays light. The app runs without these: each feature degrades gracefully
# with a "pip install ..." hint instead of crashing.

# ── OCR — searchable PDF + scanned-page text for the Translate tab ────────────
#    Pure pip, no separate native installer: the base (Chinese+English) model
#    ships inside the wheel. For accurate Italian/French/German/Spanish
#    recognition, vendor the "latin" model once (see vendor/rapidocr/README.md).
rapidocr-onnxruntime>=1.3.0,<2

# ── PDF translation — offline neural MT ───────────────────────────────────────
#    Language packages are downloaded once from "Gestisci lingue" (needs
#    internet that one time); translation itself never goes online afterwards.
#    Floor pinned at 1.11: from 1.11 argostranslate uses the lightweight
#    `minisbd` for sentence segmentation. Older 1.9.x/1.10.x instead pulled in
#    `stanza` — which (a) drags in torch + huggingface-hub and (b) was flagged
#    by pip-audit (CVE-2026-54499). Keeping the floor at 1.11 holds the resolver
#    on the lean, stanza-free engine. Do NOT lower it. If a stray `stanza` is
#    still installed from an older build, remove it: pip uninstall -y stanza
argostranslate>=1.11.0,<2

# ── Translation source cleanup ────────────────────────────────────────────────
#    Splits OCR-glued English words ("POWERUNITS" -> "POWER UNITS") before
#    translation. MIT-licensed, fully offline, ships its own word list.
wordninja>=2.0.0,<3
